# NexDynIPC 自动微分扩展设计方案

## 1. 概述

本文档详细描述了如何扩展 NexDynIPC 项目的自动微分功能，添加**反向模式自动微分（Reverse Mode AD）**和**更多数学函数支持**，使其功能更接近 JAX。

### 1.1 设计目标

1. **反向模式自动微分**：高效计算多输入单输出函数的梯度
2. **扩展数学函数库**：支持更多初等函数和特殊函数
3. **保持兼容性**：与现有前向模式 AD 共存
4. **性能优化**：表达式模板、缓存优化

### 1.2 反向模式 vs 前向模式

| 特性 | 前向模式 (DScalar1/2) | 反向模式 (新增) |
|------|----------------------|----------------|
| 适用场景 | 少输入多输出 (f: ℝⁿ→ℝᵐ, n小) | 多输入单输出 (f: ℝⁿ→ℝ, n大) |
| 内存使用 | 低 | 高（需存储计算图） |
| 计算复杂度 | O(n) | O(1) 梯度计算 |
| 实现复杂度 | 简单 | 复杂（需构建计算图） |

## 2. 系统架构

### 2.1 整体架构

```
┌─────────────────────────────────────────────────────────────────┐
│                    AutoDiff System                              │
├─────────────────────────────────────────────────────────────────┤
│  ┌──────────────┐    ┌──────────────┐    ┌──────────────┐      │
│  │ Forward Mode │    │ Reverse Mode │    │  Math Funcs  │      │
│  │  DScalar1/2  │    │   RScalar    │    │   Extended   │      │
│  └──────────────┘    └──────────────┘    └──────────────┘      │
├─────────────────────────────────────────────────────────────────┤
│                    Expression Graph                              │
│  (计算图用于反向模式，支持动态构建和前向/反向遍历)                  │
└─────────────────────────────────────────────────────────────────┘
```

### 2.2 计算图结构

```
正向计算（前向传播）:
    x₁ ──┐
         ├──► (+) ──► (*) ──► sin ──► y
    x₂ ──┘         ▲
    x₃ ────────────┘

反向计算（反向传播）:
    ∂y/∂y = 1
         │
         ▼
    sin' = cos(y_prev) * ∂y/∂y
         │
         ▼
    (*)': [∂y/∂a, ∂y/∂b]
         │
         ▼
    (+)': [∂y/∂x₁, ∂y/∂x₂]
```

## 3. 核心组件设计

### 3.1 计算图节点（Node）

```cpp
// include/NexDynIPC/Math/AutoDiff/Node.h
#pragma once

#include <Eigen/Core>
#include <memory>
#include <vector>
#include <functional>

namespace NexDynIPC::Math::AutoDiff {

/**
 * @brief 计算图节点基类
 * 
 * 每个节点代表一个运算操作，记录：
 * - 输入节点（父节点）
 * - 输出值
 * - 局部梯度（输出对输入的偏导）
 * - 伴随值（反向传播时累积）
 */
class Node : public std::enable_shared_from_this<Node> {
public:
    using SharedPtr = std::shared_ptr<Node>;
    using WeakPtr = std::weak_ptr<Node>;

    enum class OpType {
        Variable,    // 叶子节点（输入变量）
        Constant,    // 常量
        Add,         // 加法
        Sub,         // 减法
        Mul,         // 乘法
        Div,         // 除法
        Pow,         // 幂运算
        Exp,         // 指数
        Log,         // 对数
        Sqrt,        // 平方根
        Sin,         // 正弦
        Cos,         // 余弦
        Tan,         // 正切
        Asin,        // 反正弦
        Acos,        // 反余弦
        Atan,        // 反正切
        Atan2,       // 双参数反正切
        Sinh,        // 双曲正弦
        Cosh,        // 双曲余弦
        Tanh,        // 双曲正切
        Abs,         // 绝对值
        Max,         // 最大值
        Min,         // 最小值
        Sigmoid,     // Sigmoid函数
        ReLU,        // ReLU激活函数
        Custom       // 自定义函数
    };

    Node(OpType op, const std::vector<SharedPtr>& inputs = {});
    virtual ~Node() = default;

    // 前向传播：计算节点值
    virtual double forward() = 0;

    // 反向传播：计算并传播梯度
    // adjoint: 输出节点的伴随值（∂y/∂output）
    virtual void backward(double adjoint) = 0;

    // 获取节点值
    double value() const { return value_; }
    void setValue(double v) { value_ = v; }

    // 获取/设置伴随值
    double adjoint() const { return adjoint_; }
    void addAdjoint(double adj) { adjoint_ += adj; }
    void zeroAdjoint() { adjoint_ = 0.0; }

    // 获取操作类型
    OpType opType() const { return op_type_; }

    // 获取输入节点
    const std::vector<SharedPtr>& inputs() const { return inputs_; }

    // 获取输出节点（用于构建反向传播图）
    const std::vector<WeakPtr>& outputs() const { return outputs_; }
    void addOutput(const SharedPtr& output);

    // 检查是否为叶子节点
    bool isLeaf() const { return inputs_.empty(); }

    // 获取节点ID（用于调试）
    size_t id() const { return id_; }

protected:
    OpType op_type_;
    std::vector<SharedPtr> inputs_;   // 输入节点（强引用）
    std::vector<WeakPtr> outputs_;    // 输出节点（弱引用，避免循环引用）
    
    double value_ = 0.0;      // 前向值
    double adjoint_ = 0.0;    // 反向伴随值（累积梯度）
    
    size_t id_ = 0;
    static size_t next_id_;
};

/**
 * @brief 变量节点（叶子节点）
 */
class VariableNode : public Node {
public:
    explicit VariableNode(double value = 0.0);

    double forward() override;
    void backward(double adjoint) override;

    void setValue(double v) { value_ = v; }
};

/**
 * @brief 常量节点
 */
class ConstantNode : public Node {
public:
    explicit ConstantNode(double value);

    double forward() override;
    void backward(double adjoint) override;
};

} // namespace NexDynIPC::Math::AutoDiff
```

// include/NexDynIPC/Math/AutoDiff/RScalar.h
#pragma once

#include "NexDynIPC/Math/AutoDiff/Node.h"
#include <memory>
#include <functional>

namespace NexDynIPC::Math::AutoDiff {

/**
 * @brief 反向模式自动微分标量
 * 
 * 使用方式：
 * ```cpp
 * // 1. 创建变量
 * RScalar x(2.0);
 * RScalar y(3.0);
 * 
 * // 2. 构建计算图
 * RScalar z = x * x + y * y;
 * 
 * // 3. 反向传播
 * z.backward();
 * 
 * // 4. 获取梯度
 * double dz_dx = x.grad();  // 4.0
 * double dz_dy = y.grad();  // 6.0
 * ```
 */
class RScalar {
public:
    // 构造函数
    RScalar();
    explicit RScalar(double value);
    explicit RScalar(Node::SharedPtr node);

    // 获取值
    double value() const;
    
    // 获取梯度（反向传播后）
    double grad() const;

    // 反向传播入口
    void backward();
    void backward(double seed);  // 带初始种子值

    // 清零梯度
    void zeroGrad();

    // 获取底层节点
    Node::SharedPtr node() const { return node_; }

    // 算术运算符
    RScalar operator+(const RScalar& other) const;
    RScalar operator-(const RScalar& other) const;
    RScalar operator*(const RScalar& other) const;
    RScalar operator/(const RScalar& other) const;
    RScalar operator-() const;

    RScalar& operator+=(const RScalar& other);
    RScalar& operator-=(const RScalar& other);
    RScalar& operator*=(const RScalar& other);
    RScalar& operator/=(const RScalar& other);

    // 与标量的运算
    RScalar operator+(double scalar) const;
    RScalar operator-(double scalar) const;
    RScalar operator*(double scalar) const;
    RScalar operator/(double scalar) const;

    friend RScalar operator+(double scalar, const RScalar& rs);
    friend RScalar operator-(double scalar, const RScalar& rs);
    friend RScalar operator*(double scalar, const RScalar& rs);
    friend RScalar operator/(double scalar, const RScalar& rs);

    // 比较运算符
    bool operator<(const RScalar& other) const { return value() < other.value(); }
    bool operator>(const RScalar& other) const { return value() > other.value(); }
    bool operator==(const RScalar& other) const { return value() == other.value(); }
    bool operator!=(const RScalar& other) const { return value() != other.value(); }

private:
    Node::SharedPtr node_;
};

### 3.3 具体节点实现示例

```cpp
// src/Math/AutoDiff/Nodes/BinaryOpNodes.cpp

#include "NexDynIPC/Math/AutoDiff/Node.h"
#include <cmath>

namespace NexDynIPC::Math::AutoDiff {

// ==================== 加法节点 ====================
class AddNode : public Node {
public:
    AddNode(const SharedPtr& a, const SharedPtr& b) : Node(OpType::Add, {a, b}) {}

    double forward() override {
        value_ = inputs_[0]->value() + inputs_[1]->value();
        return value_;
    }

    void backward(double adjoint) override {
        // ∂(a+b)/∂a = 1, ∂(a+b)/∂b = 1
        addAdjoint(adjoint);
        inputs_[0]->backward(adjoint);
        inputs_[1]->backward(adjoint);
    }
};

// ==================== 乘法节点 ====================
class MulNode : public Node {
public:
    MulNode(const SharedPtr& a, const SharedPtr& b) : Node(OpType::Mul, {a, b}) {}

    double forward() override {
        value_ = inputs_[0]->value() * inputs_[1]->value();
        return value_;
    }

    void backward(double adjoint) override {
        // ∂(a*b)/∂a = b, ∂(a*b)/∂b = a
        addAdjoint(adjoint);
        double a_val = inputs_[0]->value();
        double b_val = inputs_[1]->value();
        inputs_[0]->backward(adjoint * b_val);
        inputs_[1]->backward(adjoint * a_val);
    }
};

// ==================== Sin节点 ====================
class SinNode : public Node {
public:
    explicit SinNode(const SharedPtr& x) : Node(OpType::Sin, {x}) {}

    double forward() override {
        value_ = std::sin(inputs_[0]->value());
        return value_;
    }

    void backward(double adjoint) override {
        // d(sin(x))/dx = cos(x)
        addAdjoint(adjoint);
        double cos_val = std::cos(inputs_[0]->value());
        inputs_[0]->backward(adjoint * cos_val);
    }
};

// ==================== Sigmoid节点 ====================
class SigmoidNode : public Node {
public:
    explicit SigmoidNode(const SharedPtr& x) : Node(OpType::Sigmoid, {x}) {}

    double forward() override {
        double x_val = inputs_[0]->value();
        // 数值稳定的sigmoid
        if (x_val >= 0) {
            double z = std::exp(-x_val);
            value_ = 1.0 / (1.0 + z);
        } else {
            double z = std::exp(x_val);
            value_ = z / (1.0 + z);
        }
        return value_;
    }

    void backward(double adjoint) override {
        // d(sigmoid(x))/dx = sigmoid(x) * (1 - sigmoid(x))
        addAdjoint(adjoint);
        double sig = value_;
        double grad = sig * (1.0 - sig);
        inputs_[0]->backward(adjoint * grad);
    }
};

// ==================== ReLU节点 ====================
class ReLUNode : public Node {
public:
    explicit ReLUNode(const SharedPtr& x) : Node(OpType::ReLU, {x}) {}

    double forward() override {
        double x_val = inputs_[0]->value();
        value_ = x_val > 0.0 ? x_val : 0.0;
        return value_;
    }

    void backward(double adjoint) override {
        // d(relu(x))/dx = 1 if x > 0 else 0
        addAdjoint(adjoint);
        double grad = inputs_[0]->value() > 0.0 ? 1.0 : 0.0;
        inputs_[0]->backward(adjoint * grad);
    }
};

} // namespace NexDynIPC::Math::AutoDiff
```

```cpp
// include/NexDynIPC/Math/AutoDiff/ComputationGraph.h
#pragma once

#include "NexDynIPC/Math/AutoDiff/Node.h"
#include <vector>
#include <stack>
#include <unordered_set>

namespace NexDynIPC::Math::AutoDiff {

/**
 * @brief 计算图管理器
 * 
 * 负责：
 * 1. 拓扑排序
 * 2. 前向传播执行
 * 3. 反向传播执行
 * 4. 内存管理优化
 */
class ComputationGraph {
public:
    /**
     * @brief 从输出节点构建计算图
     */
    explicit ComputationGraph(const Node::SharedPtr& output);

    /**
     * @brief 执行前向传播
     * @return 输出值
     */
    double forward();

    /**
     * @brief 执行反向传播
     * @param seed 初始梯度（通常为1.0）
     */
    void backward(double seed = 1.0);

    /**
     * @brief 清零所有节点的伴随值
     */
    void zeroGradients();

    /**
     * @brief 获取拓扑排序后的节点列表
     */
    const std::vector<Node::SharedPtr>& topologicalOrder() const {
        return topological_order_;
    }

    /**
     * @brief 获取所有叶子节点（输入变量）
     */
    std::vector<Node::SharedPtr> getLeafNodes() const;

    /**
     * @brief 打印计算图（用于调试）
     */
    void printGraph() const;

private:
    Node::SharedPtr output_;
    std::vector<Node::SharedPtr> topological_order_;
    std::vector<Node::SharedPtr> reverse_order_;

    // 拓扑排序（Kahn算法）
    void topologicalSort();
    
    // 收集所有节点
    void collectNodes();
};
```

### 4.2 自定义函数注册

```cpp
// include/NexDynIPC/Math/AutoDiff/CustomFunction.h
#pragma once

#include "NexDynIPC/Math/AutoDiff/RScalar.h"
#include <functional>

namespace NexDynIPC::Math::AutoDiff {

/**
 * @brief 自定义函数注册器
 * 
 * 允许用户注册自己的函数及其导数
 */
class CustomFunctionRegistry {
public:
    using ForwardFunc = std::function<double(double)>;
    using BackwardFunc = std::function<double(double)>;  // 返回导数

    /**
     * @brief 注册单输入单输出函数
     * @param name 函数名
     * @param forward 前向计算
     * @param backward 导数计算
     */
    static void registerFunction(
        const std::string& name,
        ForwardFunc forward,
        BackwardFunc backward
    );

    /**
     * @brief 创建已注册函数的RScalar版本
     */
    static RScalar call(const std::string& name, const RScalar& x);

    /**
     * @brief 检查函数是否已注册
     */
    static bool isRegistered(const std::string& name);
};

/**
 * @brief 宏定义简化自定义函数创建
 */
#define REGISTER_AUTODIFF_FUNCTION(name, forward_expr, backward_expr) \
    namespace { \
        struct name##_registrar { \
            name##_registrar() { \
                CustomFunctionRegistry::registerFunction( \
                    #name, \
                    [](double x) -> double { return forward_expr; }, \
                    [](double x) -> double { return backward_expr; } \
                ); \
            } \
        } name##_instance; \
    }

} // namespace NexDynIPC::Math::AutoDiff
```

### 4.3 性能优化：表达式模板

```cpp
// include/NexDynIPC/Math/AutoDiff/ExpressionTemplate.h
#pragma once

#include "NexDynIPC/Math/AutoDiff/RScalar.h"

namespace NexDynIPC::Math::AutoDiff {

/**
 * @brief 表达式模板基类
 * 
 * 用于延迟计算，避免临时对象
 */
template<typename Derived>
class Expression {
public:
    const Derived& derived() const {
        return static_cast<const Derived&>(*this);
    }

    double eval() const {
        return derived().eval();
    }
};

/**
 * @brief 二元运算表达式模板
 */
template<typename LHS, typename RHS, typename Op>
class BinaryExpr : public Expression<BinaryExpr<LHS, RHS, Op>> {
public:
    BinaryExpr(const LHS& lhs, const RHS& rhs) : lhs_(lhs), rhs_(rhs) {}

    double eval() const {
        return Op::apply(lhs_.eval(), rhs_.eval());
    }

private:
    const LHS& lhs_;
    const RHS& rhs_;
};

// 运算标签
struct AddOp {
    static double apply(double a, double b) { return a + b; }
};

struct MulOp {
    static double apply(double a, double b) { return a * b; }
};

/**
 * @brief 使用表达式模板的轻量级标量
 * 
 * 适用于简单表达式，避免构建完整计算图
 */
class EScalar {
public:
    template<typename Expr>
    EScalar(const Expression<Expr>& expr) : value_(expr.eval()) {}

    double value() const { return value_; }

private:
    double value_;
};

} // namespace NexDynIPC::Math::AutoDiff
```

### 6.2 神经网络激活函数

```cpp
#include <NexDynIPC/Math/AutoDiff/RScalar.h>

using namespace NexDynIPC::Math::AutoDiff;

// 定义神经网络层
RScalar neural_network_layer(const RVector& input, 
                              const RMatrix& weights,
                              const RVector& bias) {
    RVector pre_activation = weights * input + bias;
    
    // 应用激活函数
    RVector activated(pre_activation.size());
    for (size_t i = 0; i < pre_activation.size(); ++i) {
        activated[i] = relu(pre_activation[i]);  // 或 gelu, swish 等
    }
    
    // 输出层（softmax）
    return softmax(activated);
}

// 训练时使用反向传播计算梯度
void train_step(RVector& input, 
                RMatrix& weights,
                RVector& bias,
                double learning_rate) {
    RScalar loss = compute_loss(input, weights, bias);
    
    // 反向传播
    loss.backward();
    
    // 更新参数
    for (size_t i = 0; i < weights.rows(); ++i) {
        for (size_t j = 0; j < weights.cols(); ++j) {
            double grad = weights(i, j).grad();
            weights(i, j) = RScalar(weights(i, j).value() - learning_rate * grad);
        }
    }
    
    // 清零梯度
    loss.zeroGrad();
}
```

### 6.3 物理仿真中的能量最小化

```cpp
#include <NexDynIPC/Math/AutoDiff/RVector.h>

using namespace NexDynIPC::Math::AutoDiff;

// 定义弹性势能
RScalar elastic_energy(const RVector& positions, 
                       const Eigen::MatrixXi& edges,
                       double stiffness) {
    RScalar energy(0.0);
    
    for (int e = 0; e < edges.rows(); ++e) {
        int i = edges(e, 0);
        int j = edges(e, 1);
        
        RVector xi = positions.segment<3>(i * 3);
        RVector xj = positions.segment<3>(j * 3);
        
        RScalar rest_length = ...;  // 静息长度
        RScalar current_length = (xj - xi).norm();
        
        // 弹簧势能: 0.5 * k * (L - L₀)²
        RScalar stretch = current_length - rest_length;
        energy += 0.5 * stiffness * stretch * stretch;
    }
    
    return energy;
}

// 使用牛顿法最小化能量
void minimize_energy(Eigen::VectorXd& positions,
                     const Eigen::MatrixXi& edges) {
    const int max_iters = 100;
    
    for (int iter = 0; iter < max_iters; ++iter) {
        // 转换为自动微分变量
        RVector pos_ad(positions);
        
        // 计算能量和梯度
        RScalar energy = elastic_energy(pos_ad, edges, 1000.0);
        energy.backward();
        
        Eigen::VectorXd grad = pos_ad.gradients();
        
        // 检查收敛
        if (grad.norm() < 1e-6) break;
        
        // 线搜索更新位置
        double step_size = line_search(positions, grad);
        positions -= step_size * grad;
    }
}
```
